sudo apt-get purge kubeadm kubectl kubelet kubernetes-cni (remove kubernete )

docker system prune -a -f (remove docker)
sudo iptables -F && sudo iptables -t nat -F && sudo iptables -t mangle -F && sudo iptables -X
